\section{Create student table(\underline{ID}, name, major, city, state)}
\begin{lstlisting}
create table student(
  ID int,
  name varchar(20),
  major varchar(20),
  city varchar(20),
  state char(2),
  primary key(ID)
  )
\end{lstlisting}

\begin{lstlisting}
create table department(
  dnumber int,
  dept_name varchar(20) unique,
  building varchar(20),
  primary key(dnumber)
  )
\end{lstlisting}
